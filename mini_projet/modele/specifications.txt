Data Mart Travel :
==================
Fact : Travel
  - id_travel : int (degenerate dimension)
  - id_date : int
  - id_time : int
  - id_traveler : int
  - id_line : int
  - id_station : int
  - id_vehicle : int
  - travel_cost : real (in euros)

Dimension : Line
  - id : int
  - num_line : int
  - type_vehicle : string (bus or tram)
  - start_station : string
  - end_station : string
  - nb_stations : int
  - distance_travel : int (in meters)
  - avg_duration_travel : int (in minutes)
  - avg_duration_btw_stations : int (in minutes)
  - avg_served_people : int

Dimension : Station
  - id : int
  - name : string
  - coord_x : float
  - coord_y : float
  - place : string
  - disability_access : bool
  - sheltered_for_rain : bool

Dimension : Vehicle
  - id : int
  - type : string
  - model : string
  - max_speed : int (in km/h)
  - capacity_max : int (theoretical maximum number people total in vehicule)
  - nb_seats : int
  - wheelchair_capacity : int (number of seat for handicapped people)

Dimension : Traveler
  - id : int
  - anonymous : bool (true if the traveler is subscribed, false otherwise)
  - name : string
  - surname : string
  - birth_year : int
  - gender : string
  - address : string
  - nationality : string
  - subscription_type : string
  - subscription_fees : float
  - ticket_price : float (only if the traveler is not subscribed)

Dimension : Date
  - id : int
  - date : Date
  - year : int
  - month_year : int
  - month_calendar : string
  - day_month : int
  - day_calendar : string
  - day_week : int
  - day_year : int
  - holiday_indicator : string
  - weekday_indicator : string

Dimension : Time
  - id : int
  - full_time_description : string
  - hours : int
  - minutes : int
  - seconds : int
  - AM_PM_indicator : string
  - day_part_segment : string

Data Mart Maintenance :
=======================
Fact : Maintenance
  - id_date : int
  - id_time : int
  - id_vehicle : int
  - id_driver : int
  - id_technician : int
  - id_technical_area : int
  - cost : float
  - estimated_time : int (in minutes)

Dimension : Date (shared dimension between data marts)
Dimension : Time (shared dimension between data marts)
Dimension : Vehicle (shared dimension between data marts)
Dimension : Driver
  - id : int
  - name : string
  - surname : string
  - birth_year : int
  - gender : string
  - address : string
  - nationality : string
  - contract_type : string
  - contract_start_year : int
  - salary : float
  - experience_years : int
  - tram_driver : bool
  - bus_driver : bool

Dimension : Technician
  - id : int
  - name : string
  - surname : string
  - birth_year : int
  - gender : string
  - address : string
  - nationality : string
  - contract_type : string
  - contract_start_year : int
  - salary : float
  - experience_years : int
  - speciality : string

Dimension : TechnicalArea
  - id : int
  - address : string
  - type_maintained_vehicles : string (bus, tram, both)
  - vehicles_capacity : int
  - surface : float (en m^2)
  - monthly_costs : float
  - equipment_level : int (entre 1 pour pas assez équipé et 5 pour assez équipé)
  - workforce_size : int (effectif)
